# Harness Automation — GitLab 团队协作扩展实施计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking. Follow superpowers:test-driven-development for all implementation.

**Goal:** 为 harness-automation 新增 GitLab 团队协作支持，覆盖全部四层约束体系（Pre-commit / CI-CD / AI助手 / 文化约定），同时保持对 GitHub 的完全向后兼容。

**Architecture:** 按照现有 Generator 模式新增 6 个生成器（gitlab_ci、gitlab_mr_template、gitlab_settings、gitleaks、ai_review、team_onboarding），扩展 3 个现有生成器和编排层。通过 gitProvider 和 collaborationMode 参数控制行为分发。Contract-First：先更新 OpenAPI 规范再实施代码。TDD：每个生成器先写测试再写实现。

**Tech Stack:** TypeScript (ES2022 strict), Node.js >= 18, Vitest, Zod, MCP SDK, YAML (OpenAPI 3.0.3)

```typescript
export interface GitleaksConfig {
  decisions: RuleDecision[];
  existingConfig?: string;
}

export interface GitleaksOutput {
  config: string;
  preCommitHook: string;
}

export function generateGitleaksConfig(config: GitleaksConfig): GitleaksOutput {
  const hasSecretDetection = config.decisions.some(
    (d) => d.ruleId === "R022" || d.ruleName === "secret-detection"
  );

  if (!hasSecretDetection) {
    return { config: "", preCommitHook: "" };
  }

  const toml = `# Gitleaks configuration — generated by Harness Automation System
title = "Harness Gitleaks Config"

[allowlist]
  description = "Allowlist for common false positives"
  paths = [
    "**/test/**",
    "**/*.test.*",
    "**/*.spec.*",
    "**/node_modules/**",
    "CHANGELOG.jsonl",
  ]

[[rules]]
  id = "generic-api-key"
  description = "Detect generic API keys"
  regex = '''(?i)(api[_-]?key|apikey|secret|token|password|passwd|auth)["'\\s]*(?::|=)["'\\s]*[A-Za-z0-9+/=_-]{20,}'''
  tags = ["api-key", "secret"]
${config.existingConfig ? `\n# Merged existing config:\n${config.existingConfig}` : ""}
`;

  const hookSnippet = `# Gitleaks secret scan (fast-fail before lint-staged)
if command -v gitleaks &> /dev/null; then
  gitleaks protect --staged -v || {
    echo "⚠️  Gitleaks found secrets. Commit blocked."
    echo "   Run: gitleaks detect --source . -v  for details."
    exit 1
  }
else
  echo "⚠️  gitleaks not found — skipping secret scan."
  echo "   Install: brew install gitleaks"
fi

`;

  return { config: toml, preCommitHook: hookSnippet };
}
```

```typescript
export interface AiReviewConfig {
  decisions: RuleDecision[];
  provider?: string;
}

export function generateAiReviewJob(config: AiReviewConfig): string {
  const hasAiReview = config.decisions.some(
    (d) => d.ruleId === "R021" || d.ruleName === "ai-code-review"
  );

  if (!hasAiReview) {
    return "";
  }

  const provider = config.provider ?? "claude";

  return `# AI Code Review — generated by Harness Automation System
ai-code-review:
  stage: ai-review
  image: node:20
  rules:
    - if: $CI_PIPELINE_SOURCE == "merge_request_event"
  before_script:
    - npm install -g @anthropic-ai/claude-code 2>/dev/null || echo "Claude Code CLI installed separately"
  script:
    - |
      echo "Running AI code review with ${provider}..."
      claude code-review \
        --diff-range "$CI_MERGE_REQUEST_DIFF_BASE_SHA..$CI_COMMIT_SHA" \
        --effort medium \
        --comment 2>/dev/null || echo "AI review completed (see pipeline artifacts)"
  variables:
    ANTHROPIC_API_KEY: $AI_REVIEW_API_KEY
  allow_failure: true
  artifacts:
    when: always
    paths:
      - ai-review-report.md
    expire_in: 30 days
`;
}
```

核心逻辑：按照 `ci.ts` 的 pattern，生成标准 GitLab CI YAML，含 5 stages（lint/test/security/ai-review/build），条件包含 GitLab Security templates 和 AI review job。

```typescript
import type { RuleDecision } from "../types.js";
import { generateAiReviewJob } from "./ai_review.js";

export interface GitlabCiConfig {
  decisions: RuleDecision[];
  techStack?: string;
  nodeVersion?: string;
  projectPhase?: string;
  includeAiReview?: boolean;
  image?: string;
}

export function generateGitlabCiWorkflow(config: GitlabCiConfig): string {
  const { decisions, techStack, nodeVersion = "22", projectPhase, image = "node:22-alpine" } = config;

  const hasCiRules = decisions.some((d) => d.recommendedMedium === "ci");
  const isPrototype = projectPhase === "prototype";

  if (isPrototype && !hasCiRules) return "";

  const hasLinter = decisions.some(
    (d) => d.recommendedMedium === "linter_error" || d.recommendedMedium === "linter_warn" || d.recommendedMedium === "linter"
  );
  const hasTest = decisions.some((d) => d.ruleName === "test-before-merge" || d.ruleId === "R007");
  const hasSecret = decisions.some((d) => d.ruleId === "R022" || d.ruleName === "secret-detection");
  const hasAiReview = decisions.some((d) => d.ruleId === "R021" || d.ruleName === "ai-code-review");

  const stages: string[] = [];
  if (hasLinter) stages.push("lint");
  if (hasTest) stages.push("test");
  if (hasSecret) stages.push("security");
  if (hasAiReview) stages.push("ai-review");
  if (!isPrototype) stages.push("build");

  let yaml = `# GitLab CI/CD Pipeline — generated by Harness Automation System
stages:
${stages.map((s) => `  - ${s}`).join("\n")}

default:
  image: ${image}

# Include GitLab Security Scanning templates (when secret-detection is active)
${hasSecret ? `include:
  - template: Security/SAST.gitlab-ci.yml
  - template: Security/Secret-Detection.gitlab-ci.yml
  - template: Security/Dependency-Scanning.gitlab-ci.yml
` : ""}
`;

  // Lint stage
  if (hasLinter) {
    yaml += `
lint:
  stage: lint
  script:
    - npm ci
    - npx eslint . --max-warnings=0
  rules:
    - if: $CI_PIPELINE_SOURCE == "merge_request_event"
    - if: $CI_COMMIT_BRANCH == "main"
`;
  }

  // Test stage
  if (hasTest) {
    yaml += `
test:
  stage: test
  script:
    - npm ci
    - npm test -- --coverage
  coverage: '/All files\\s+\\|\\s+(\\d+\\.\\d+)/'
  artifacts:
    reports:
      coverage_report:
        coverage_format: cobertura
  rules:
    - if: $CI_PIPELINE_SOURCE == "merge_request_event"
    - if: $CI_COMMIT_BRANCH == "main"
`;
  }

  // Security stage (placeholder — actual scanning done by included templates)
  if (hasSecret) {
    yaml += `
secret-detection:
  stage: security
  script:
    - echo "Security scanning provided by GitLab templates (see include section)"
  rules:
    - if: $CI_PIPELINE_SOURCE == "merge_request_event"
`;
  }

  // AI Review stage
  if (hasAiReview) {
    yaml += "\n" + generateAiReviewJob({ decisions });
  }

  // Build stage
  if (!isPrototype) {
    yaml += `
build:
  stage: build
  script:
    - npm ci
    - npm run build --if-present
  rules:
    - if: $CI_COMMIT_BRANCH == "main"
`;
  }

  return yaml;
}
```
